using System;
					
public class Program
{
	public static string ReplaceUnderscoreAndCapitalizeWord(string str) 
	{
		string[] frags = str.Split('_');
		for(int idx = 0; idx < frags.Length; ++idx)
		{
			string local = frags[idx];
			frags[idx] = string.Concat(local[0].ToString().ToUpper(), local.AsSpan(1).ToString());
		}
		
		return string.Join(string.Empty, frags);
	}    	
		

	
	public static void Main()
	{
		Console.WriteLine(ReplaceUnderscoreAndCapitalizeWord("enter_method_name_here"));
	}
}
